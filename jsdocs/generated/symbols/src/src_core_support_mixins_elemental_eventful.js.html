1 (function () {
  2     var eventSplitter = /^(\w+)\s*(.*)$/;
  3 
  4     /**
  5      * @class
  6      * @desc Resolves and assigns DOM listeners for
  7      * events defined in events property.
  8      *
  9      * @requires Ribs.mixins.support.dom.element
 10      * @requires Ribs.mixins.support.rendering.renderChain
 11      */
 12     Ribs.mixins.support.dom.eventful = {
 13         /**
 14          * @field
 15          * @desc Hash of events to listen. Key is of form
 16          * "<event> <selector>" and value is the
 17          * name of the function to call on trigger.
 18          */
 19         events: {},
 20 
 21         unbindEvents: function () {
 22             if (this.el) {
 23                 this.el.unbind();
 24             }
 25         },
 26         bindEvents: function () {
 27             if (!this.events || !this.el) {
 28                 return;
 29             }
 30             _.each(this.events, _.bind(function (methodName, key) {
 31                 var match = key.match(eventSplitter),
 32                         eventName = match[1], selector = match[2],
 33                         method = _.bind(this[methodName], this);
 34                 if (selector === '') {
 35                     this.el.bind(eventName, method);
 36                 } else {
 37                     this.el.delegate(selector, eventName, method);
 38                 }
 39             }, this));
 40         }
 41     };
 42 }());
