  4 Ribs.backbone.mixins.support.invalidating = {
  5     renderingHash: [],
  6     redrawHash: [],
  7     
  8     mixinInitialize: function () {
  9         var that = this;
 10 
 11         // Set up model change listeners for rendering only.
 12         _.each(this.renderingHash, function (attributeName, modelName) {
 13             that.models[modelName].bind("change:" + attributeName, this.render);
 14         });
 15 
 16         // Set model change listeners for invalidate+render.
 17         _.each(this.redrawHash, function (attributeName, modelName) {
 18             that.models[modelName].bind("change:" + attributeName, function () {
 19                 that.invalidated = true;
 20                 that.render();
 21             });
 22         });
 23     }
 24 };
