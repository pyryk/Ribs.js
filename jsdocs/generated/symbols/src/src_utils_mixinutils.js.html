1 /**
  2  * @method
  3  * @desc Parses given mixins and returns an array that
  4  * contains the initialized mixins. For syntax see samples.
  5  *
  6  * @param mixinDefinitions An Array or Object that contains
  7  * the definitions. An array is assumed to contain a list of
  8  * individual objects of { <mixinName>: <options> } and an
  9  * Object { <elementSelector>: [ ... ] }.
 10  */
 11 Ribs.parseMixinDefinitions = function (mixinDefinitions) {
 12     mixinDefinitions = mixinDefinitions || [];
 13     var mixinClasses = [], i, l,
 14         processMixinDefinition = function (options, name) {
 15             var mixinFunction = Ribs.mixins[name];
 16             if (!mixinFunction) {
 17                 throw "Could not find mixin " + name;
 18             }
 19             mixinClasses.push(mixinFunction(options));
 20         };
 21 
 22     if (_.isArray(mixinDefinitions)) {
 23         for (i = 0, l = mixinDefinitions.length; i < l; i++) {
 24             var mixinDefinitionObject = mixinDefinitions[i];
 25             _.each(mixinDefinitionObject, processMixinDefinition);
 26         }
 27     } else {
 28         _.each(mixinDefinitions,
 29             function (nestedMixinDefinitionArray, elementSelector) {
 30                 var MixinComposite = Ribs.mixins.mixinComposite({
 31                     mixins: nestedMixinDefinitionArray,
 32                     elementSelector: elementSelector
 33                 });
 34                 mixinClasses.push(MixinComposite);
 35             }
 36         );
 37     }
 38     return mixinClasses;
 39 };
