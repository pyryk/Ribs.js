1 describe("Ribs.support.mixins.parsing", function () {
  2     var mixin, parser, def, composite, customMixin;
  3 
  4     beforeEach(function () {
  5         mixin = _.clone(Ribs.support.mixins.parsing);
  6         customMixin = {};
  7         parser = Ribs.utils.createMixinDefinitionParser({
  8             mixinLibrary: {
  9                 plain: { param1: "foo"},
 10                 composite: Ribs.mixins.composite,
 11                 customNameSpace: {
 12                     actualMixin: customMixin
 13                 }
 14             }
 15         });
 16     });
 17 
 18     describe("Parsing an array", function () {
 19         beforeEach(function () {
 20             def = [
 21                 { plain: { elementSelector: ".mySel" }},
 22                 { plain: { param1: "testparam" }},
 23                 { plain: {}}
 24             ];
 25             composite = _.clone(Ribs.support.mixins.composite);
 26             parser.createCompositeFromDefinitions({
 27                 mixinDefinitions: def,
 28                 composite: composite
 29             });
 30         });
 31 
 32         it("Should consider elements of an array as mixins", function () {
 33             expect(composite).toHaveNumberOfchildrenTypes(def.length);
 34         });
 35 
 36         it("Should consider the key in a mixin definition as the base, and the value as the overrides", function () {
 37             expect(composite.childrenTypes[0].elementSelector).toEqual(".mySel");
 38             expect(composite.childrenTypes[1].param1).toEqual("testparam");
 39             expect(composite.childrenTypes[2].param1).toEqual("foo");
 40         });
 41     });
 42 
 43     describe("Parsing an object", function () {
 44         beforeEach(function () {
 45             def = {
 46                 "": [],
 47                 "elementOne": [],
 48                 "myEl": []
 49             };
 50             composite = parser.createCompositeFromDefinitions({ mixinDefinitions: def });
 51         });
 52 
 53         it("Should be able to read elementSelector from the high level object keys", function () {
 54             expect(composite).toContainOneMixinWithElementSelector("");
 55             expect(composite).toContainOneMixinWithElementSelector("elementOne");
 56             expect(composite).toContainOneMixinWithElementSelector("myEl");
 57             expect(composite).toHaveNumberOfchildrenTypes(3);
 58         });
 59     });
 60 
 61     describe("Parsing an object with nested namespace", function () {
 62         beforeEach(function () {
 63             def = [{"customNameSpace.actualMixin": {}}];
 64             composite = parser.createCompositeFromDefinitions({ mixinDefinitions: def });
 65         });
 66 
 67         it("Should find the nested mixins", function () {
 68             expect(composite.childrenTypes[0]).toEqual(customMixin);
 69         });
 70     });
 71 });
