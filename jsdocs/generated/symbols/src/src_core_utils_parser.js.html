1 /**
  2  * @method
  3  * @param parserOptions
  4  */
  5 Ribs.utils.createMixinDefinitionParser = function (parserOptions) {
  6     parserOptions = parserOptions || {};
  7 
  8     var parser = { },
  9         mixinLibrary = parserOptions.mixinLibrary || {};
 10 
 11     parser.createCompositeFromDefinitions = function (options) {
 12         options = options || {};
 13 
 14         if (!mixinLibrary.hasOwnProperty("composite")) {
 15             Ribs.throwError("noCompositeMixinFoundForParsing");
 16         }
 17 
 18         var composite = options.composite || _.clone(mixinLibrary.composite),
 19             mixinDefinitions = options.mixinDefinitions || [],
 20             mixinClasses = options.mixinClasses || [];
 21 
 22         composite.elementSelector = options.elementSelector;
 23 
 24         if (_.isArray(mixinDefinitions)) {
 25             var parseOne = function (value, key) {
 26                 try {
 27                     var mixin = Ribs.utils.findObject(mixinLibrary, key);
 28                 } catch (e) {
 29                     Ribs.throwError("mixinTypeNotFound", key);
 30                 }
 31                 mixinClasses.push(_.extend({}, mixin, value));
 32             };
 33             for (var i = 0; i < mixinDefinitions.length; i++) {
 34                 _.each(mixinDefinitions[i], parseOne);
 35             }
 36         } else {
 37             var _createMixinFromDefinitions = function (nestedMixinDefinitions, elementSelector) {
 38                 var mixin = parser.createCompositeFromDefinitions({
 39                     mixinDefinitions: nestedMixinDefinitions,
 40                     elementSelector: elementSelector
 41                 });
 42                 mixinClasses.push(mixin);
 43             };
 44             _.each(mixinDefinitions, _createMixinFromDefinitions);
 45         }
 46         composite.mixinClasses = mixinClasses;
 47         return composite;
 48     };
 49 
 50     return parser;
 51 };
 52 
 53 /**
 54  * @field
 55  */
 56 Ribs.mixinParser = Ribs.utils.createMixinDefinitionParser({ mixinLibrary: Ribs.mixins });
