1 /**
  2  * @class
  3  * @requires Ribs.mixins.support.parent
  4  * @requires Ribs.mixins.support.dom.element
  5  * @requires Ribs.mixins.support.rendering.renderChain
  6  */
  7 Ribs.mixins.support.childrenElementResolver = {
  8     mixinInitialize: function () {
  9         this.resolveChildMixinElements();
 10     },
 11     redraw: function () {
 12         this.resolveChildMixinElements();
 13     },
 14     resolveChildMixinElements: function () {
 15         if (this.children && this.el) {
 16             var el = $(this.el).find(this.elementSelector);
 17             if (el.length === 0) {
 18                 el = this.el;
 19             }
 20             _.each(this.children, function (mixin) {
 21                 mixin.el = mixin.elementSelector ? el.find(mixin.elementSelector) : el;
 22             });
 23         }
 24     }
 25 };
