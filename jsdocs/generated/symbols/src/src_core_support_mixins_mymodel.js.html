1 /**
  2  * @class
  3  * @desc Observer a model with the given name and
  4  * calls myModelAdded and myModelRemoved whenever
  5  * applicable.
  6  */
  7 Ribs.mixins.support.modelChooser = {
  8     /**
  9      * @field
 10      * @desc Name of the model to use.
 11      */
 12     modelName: null,
 13 
 14     /**
 15      * @method
 16      * @param name Name Of the model that was added.
 17      * @param model Model that was added.
 18      */
 19     modelAdded: function (name, model) {
 20         if (name !== undefined && name === this.modelName) {
 21             this.myModel = model;
 22             if (_.isFunction(this.myModelAdded)) {
 23                 this.myModelAdded(model);
 24             }
 25         }
 26     },
 27 
 28     /**
 29      * @method
 30      * @param name Name Of the model that was removed.
 31      * @param model Model that was removed.
 32      */
 33     modelRemoved: function (name, model) {
 34         if (name !== undefined && name === this.modelName) {
 35             this.myModel = null;
 36             if (_.isFunction(this.myModelRemoved)) {
 37                 this.myModelRemoved(model);
 38             }
 39         }
 40     }
 41 };
