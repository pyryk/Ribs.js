1 describe("Ribs.mixins.templated", function () {
  2     var templateString;
  3 
  4     beforeEach(function () {
  5         templateString = "<div id=\"hello\"><span>My Text</span></div>";
  6     });
  7 
  8     describe("as a mixin", function () {
  9         var mixin;
 10 
 11         beforeEach(function () {
 12             mixin = _.extend({}, Ribs.mixins.templated, {
 13                 templateString: templateString,
 14                 el: $("<div></div>")
 15             });
 16             mixin.mixinInitialize();
 17             mixin.redraw();
 18         });
 19 
 20         it("Should create templated el inside the given el", function () {
 21             expect($(mixin.el).attr("id")).toEqual("");
 22             expect($(mixin.el.children()[0]).attr("id")).toEqual("hello");
 23             expect($(mixin.el.children()[0]).html()).toEqual("<span>My Text</span>");
 24         });
 25 
 26         it("Should use json property as the values for the template", function () {
 27             mixin.json = { qwert: "yuio" };
 28             mixin.templateFunction = _.template("<span><%= qwert %></span>");
 29             mixin.redraw();
 30             expect($(mixin.el).text()).toEqual("yuio");
 31         });
 32     });
 33 
 34     describe("redrawing with templates", function () {
 35         it("Should redraw using the nested templates", function () {
 36             var mixin = _.extend({}, Ribs.mixins.templated, {
 37                 templateSelector: "#test-tmpl",
 38                 overwriteEl: true
 39             });
 40             mixin.mixinInitialize();
 41             mixin.json = { qwert: "yuio" };
 42             mixin.redraw();
 43             expect($(mixin.el.children()[0]).text()).toEqual("yuio");
 44         });
 45     });
 46 });
