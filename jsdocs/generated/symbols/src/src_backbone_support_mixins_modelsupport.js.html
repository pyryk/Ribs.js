1 /**
  2  * @class
  3  */
  4 Ribs.backbone.mixins.support.model.modelBase = {
  5     backboneModels: null,
  6     createInternalModel: true,
  7     inheritingMethods: ["modelRemoved", "modelAdded"],
  8 
  9     mixinInitialize: function () {
 10         var backboneModels = this.backboneModels || {};
 11 
 12         if (this.createInternalModel) {
 13             backboneModels = _.extend({
 14                 internal: new Backbone.Model()
 15             }, backboneModels);
 16         }
 17 
 18         if (this.models) {
 19             this.models.set(backboneModels);
 20         } else {
 21             this.models = new Backbone.Model(backboneModels);
 22         }
 23 
 24         this.models.bind("change", _.bind(function () { this.modelChangeHandler(); }, this));
 25 
 26         _.each(this.models.attributes, _.bind(function (model, name) {
 27             this.modelAdded(name, model);
 28         }, this));
 29     },
 30 
 31     modelChangeHandler: function () {
 32         var previousAttributes = this.models.previousAttributes(),
 33             changedAttributes = this.models.changedAttributes(),
 34             broadcastChange = _.bind(function (value, key) {
 35                 if (this.models.previous(key)) {
 36                     this.modelRemoved(key, this.models.previous(key));
 37                 }
 38                 if (this.models.get(key)) {
 39                     this.modelAdded(key, this.models.get(key));
 40                 }
 41             }, this);
 42 
 43         // Backbone changedAttributes() does not include any
 44         // deleted ones. Check for them manually.
 45         for (var key in previousAttributes) {
 46             if (previousAttributes.hasOwnProperty(key) && !this.models.attributes.hasOwnProperty(key)) {
 47                 this.modelRemoved(key, this.models.previous(key));
 48             }
 49         }
 50         if (changedAttributes) {
 51             _.each(changedAttributes, broadcastChange);
 52         }
 53     },
 54 
 55     modelRemoved: function (name, oldModel) { },
 56 
 57     modelAdded: function (name, newModel) { },
 58 
 59     getModelJSON: function (options) {
 60         var json,
 61             modelName = options.jsonModelName;
 62 
 63         if (!modelName || !this.models) {
 64             return;
 65         }
 66 
 67         if (_.isArray(modelName)) {
 68             json = {};
 69             for (var i = 0; i < modelName.length; i++) {
 70                 _.extend(json, this.getModelJSON({ jsonModelName: modelName[i] }));
 71             }
 72         } else {
 73             if (!this.models.attributes.hasOwnProperty(modelName)) {
 74                 Ribs.throwError("modelNotFound", "" + modelName);
 75                 return;
 76             }
 77             json = this.models.get(modelName).toJSON();
 78         }
 79         return json;
 80     },
 81 
 82     getValue: function (options) {
 83         var modelName = options.valueModelName,
 84             attributeName = options.valueAttributeName;
 85 
 86         if (!modelName || !attributeName || !this.models) {
 87             return;
 88         }
 89 
 90         if (!this.models.attributes.hasOwnProperty(modelName)) {
 91             Ribs.throwError("modelNotFound", "" + modelName);
 92             return;
 93         }
 94         if (!this.models.get(modelName).attributes.hasOwnProperty(attributeName)) {
 95             Ribs.throwError("attributeNotFound", "model=" + modelName + ", attribute=" + attributeName);
 96             return;
 97         }
 98         return this.models.get(modelName).get(attributeName);
 99     },
100 
101     setValue: function (options) {
102         var modelName = options.valueModelName,
103             attributeName = options.valueAttributeName,
104             value = options.value,
105             newValues;
106 
107         if (!modelName || !attributeName) {
108             return;
109         }
110 
111         if (!this.models.attributes.hasOwnProperty(modelName)) {
112             Ribs.throwError("modelNotFound", modelName);
113             return;
114         }
115         newValues = {};
116         newValues[attributeName] = value;
117         this.models.get(modelName).set(newValues);
118     }
119 };
