1 /**
  2  * @method
  3  * @desc Combines a number of mixins and takes care of them.
  4  * This includes delegating methods and managing some syntax mixins use.
  5  *
  6  * @param classOptions Object parameter.
  7  * @param classOptions.elementSelector jQuery selector that is
  8  * invoked on the DOM element of the View that this mixin belongs to.
  9  * @param classOptions.mixinClasses Array of plain preinitialized mixins.
 10  * No parsing will occur for these. Overrides classOptions.mixins.
 11  * @param classOptions.mixins Array or Object of mixins with the Ribs.js
 12  * mixin notation. See samples for how this works.
 13  * @param classOptions.attributeName Name of the data attribute associated to this
 14  * mixin. It is retrieved from the original data model. If a mixin declares this
 15  * property, the value is inserted into "myValue" property of this mixin.
 16  * @param classOptions.uiAttributeName Name of the UI attribute that this mixin uses.
 17  * The corresponding value is taken from the ribsUI property of the
 18  * original model.If a mixin declares this property, the value is inserted
 19  * into "myValue" property of this mixin.
 20  */
 21 Ribs.mixins.mixinComposite = function (classOptions) {
 22     classOptions = classOptions || {};
 23 
 24     var elementSelector = classOptions.elementSelector,
 25         elementCreator = classOptions.elementCreator,
 26         mixinClasses = classOptions.mixinClasses || Ribs.parseMixinDefinitions(classOptions.mixins),
 27         callAllMixins = function (mixins, methodName, originalArguments) {
 28             _.each(mixins, function (mixin) {
 29                 if (mixin[methodName]) {
 30                     mixin[methodName].apply(mixin, originalArguments);
 31                 }
 32             });
 33         },
 34         updateMixinEl = function (mixin, el) {
 35             mixin.el = mixin.elementSelector ? el.find(mixin.elementSelector) : el;
 36         },
 37         updateMixinMyValue = function (mixin) {
 38             if (mixin.attributeName && mixin.dataModel) {
 39                 mixin.myValue = mixin.dataModel.get(mixin.attributeName);
 40             } else if (mixin.uiAttributeName) {
 41                 mixin.myValue = mixin.uiModel.get(mixin.uiAttributeName);
 42             } else {
 43                 mixin.myValue = null;
 44             }
 45         },
 46         eventSplitter = /^(\w+)\s*(.*)$/,
 47 
 48         MixinCompositeInst = function (parentView, model) {
 49             this.customInitialize = function () {
 50                 this.mixins = [];
 51                 _.each(mixinClasses, _.bind(function (MixinClass) {
 52                     var mixin = new MixinClass(parentView, model);
 53                     _.bind(function () { _.bindAll(this); }, mixin)();
 54                     mixin.uiModel = (model && model.ribsUI) || new Backbone.Model();
 55                     if (mixin.modelChanging) {
 56                         mixin.modelChanging();
 57                     }
 58                     mixin.dataModel = model;
 59                     updateMixinMyValue(mixin);
 60                     if (mixin.modelChanged) {
 61                         mixin.modelChanged(mixin.dataModel);
 62                     }
 63                     this.mixins.push(mixin);
 64                 }, this));
 65                 callAllMixins(this.mixins, "customInitialize", arguments);
 66             };
 67 
 68             this.unbindEvents = function () {
 69                 if (this.el) {
 70                     this.el.unbind();
 71                 }
 72                 _.each(this.mixins, function (mixin) {
 73                     if (mixin.el) {
 74                         mixin.el.unbind();
 75                     }
 76                     if (mixin.unbindEvents) {
 77                         mixin.unbindEvents.apply(mixin);
 78                     }
 79                 });
 80             };
 81 
 82             this.bindEvents = function () {
 83                 _.each(this.mixins, function (mixin) {
 84                     if (mixin.bindEvents) {
 85                         mixin.bindEvents.apply(mixin);
 86                     }
 87                     if (!mixin || !mixin.events || !mixin.el || !mixin.el.is(":visible")) {
 88                         return;
 89                     }
 90                     _.each(mixin.events, _.bind(function (methodName, key) {
 91                         var match = key.match(eventSplitter),
 92                             eventName = match[1], selector = match[2],
 93                             method = _.bind(this[methodName], this);
 94                         if (selector === '') {
 95                           mixin.el.bind(eventName, method);
 96                         } else {
 97                           mixin.el.delegate(selector, eventName, method);
 98                         }
 99                     }, mixin));
100                 });
101             };
102 
103             this.modelChanged = function (newModel) {
104                 _.each(this.mixins, _.bind(function (mixin) {
105                     mixin.dataModel = newModel;
106                     mixin.uiModel = newModel ? newModel.ribsUI : new Backbone.Model();
107                     updateMixinMyValue(mixin);
108                     if (mixin.modelChanged) {
109                         mixin.modelChanged.apply(mixin, [newModel]);
110                     }
111                 }, this));
112             };
113 
114             this.redraw = function (parentEl) {
115                 this.el = $(parentEl).find(elementSelector);
116                 if (this.el.length === 0) {
117                     if (elementCreator) {
118                         this.el = $(parentEl).append($(elementCreator));
119                     } else {
120                         this.el = $(parentEl);
121                     }
122                 }
123                 _.each(this.mixins, _.bind(function (mixin) {
124                     updateMixinMyValue(mixin);
125                     updateMixinEl(mixin, this.el);
126                     if (mixin.redraw) {
127                         mixin.redraw.apply(mixin, [mixin.el]);
128                     }
129                 }, this));
130             };
131 
132             this.refresh = function () {
133                 _.each(this.mixins, _.bind(function (mixin) {
134                     updateMixinMyValue(mixin);
135                     if (mixin.refresh) {
136                         mixin.refresh.apply(mixin);
137                     }
138                 }, this));
139             };
140         };
141 
142     _.each(Ribs.mixinMethods, function (methodName) {
143         if (!MixinCompositeInst.prototype.hasOwnProperty(methodName)) {
144             MixinCompositeInst.prototype[methodName] = function () {
145                 callAllMixins(this.mixins, methodName, arguments);
146             };
147         }
148     });
149 
150     return MixinCompositeInst;
151 };
