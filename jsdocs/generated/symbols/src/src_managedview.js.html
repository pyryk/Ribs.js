1 /**
  2  * @augments Backbone.View
  3  * @class A more sophisticated Backbone.View that adds
  4  * support for component lifecycles, as often needed
  5  * in RIA applications.
  6 **/
  7 Ribs.ManagedView = Backbone.View.extend(
  8 /** @lends Ribs.ManagedView */
  9 {
 10 
 11     /**
 12      * @field
 13      * @desc When calling render(), if this flag is set,
 14      * a redraw will occur. When redraw is finished, this
 15      * flag is again reset to false.
 16      */
 17     invalidated: true,
 18     
 19     /**
 20      * @field
 21      * @desc Is set when the component has finished initializing
 22      * and is ready to be rendered.
 23      */
 24     initialized: false,
 25 
 26     /**
 27      * @method
 28      * @desc Override to Backbone.View.initialize; Is called on
 29      * creation, using the "new" operator.
 30      */
 31     initialize: function () {
 32         _.bindAll(this, "customInitialize", "bindToModel", "modelChanging", "modelChanged", "render", "unbindEvents", "redraw", "refresh", "bindEvents", "hide", "dispose");
 33         Backbone.View.prototype.initialize.apply(this, arguments);
 34         if (this.model) {
 35             this.bindToModel(this.model);
 36         }
 37         this.customInitialize();
 38         this.initialized = true;
 39         this.render();
 40     },
 41 
 42     /**
 43      * @method
 44      * @desc Is called after the "real" initialize is completed,
 45      * apart from setting initialized flag and rendering. Without
 46      * the initialized flag the component will not render.<br /><br />
 47      *
 48      * This is a clean place for generic initialization code.
 49      * A call to this method is not expected of an overriding method.
 50      */
 51     customInitialize: function () { },
 52     
 53     /**
 54      * @method
 55      * @desc Backbone convention for setting/changing the model
 56      * for a View. Here is it is defined explicitly since the View
 57      * life cycle requires it. A possible old model is unbound of
 58      * its change handler, and the new one is in turn bound.<br /><br />
 59      *
 60      * If the new model does not yet augmented with UI attributes,
 61      * Ribs.augmentModelWithUIAttributes does this.
 62      *
 63      * @param model The Backbone.Model we want this View to represent.
 64      */
 65     bindToModel: function (model) {
 66         this.modelChanging();
 67         if (this.model && this.model.ribsUI) {
 68              this.model.ribsUI.safeUnbind("all", this.render);
 69         }
 70         this.model = model;
 71         if (this.model) {
 72             Ribs.augmentModelWithUIAttributes(this.model);
 73             this.model.ribsUI.bind("all", this.render);
 74         }
 75         this.invalidated = true;
 76         this.modelChanged(model);
 77     },
 78     modelChanging: function () { },
 79     modelChanged: function (newModel) { },
 80     
 81     /**
 82      * @method
 83      * @desc Override to Backbone.View.render.<br /><br />
 84      *
 85      * Exits if initialized flag is not set.<br /><br />
 86      *
 87      * Maintains the View life cycle of rendering. First
 88      * unbinds all events, then redraws if invalidated flag
 89      * is set, and after that calls refresh and binds all events.
 90      */
 91     render: function () {
 92         if (!this.initialized) { return; }
 93         this.unbindEvents();
 94         if (this.invalidated) {
 95             this.redraw(this.el);
 96             this.invalidated = false;
 97         }
 98         this.refresh();
 99         this.bindEvents();
100     },
101 
102     /**
103      * @method
104      * @desc Clear all listeners to DOM events. The default
105      * implementation only calls $(this.el).unbind().
106      */
107     unbindEvents: function () {
108         $(this.el).unbind();
109     },
110 
111     /**
112      * @method
113      * @desc Set all listeners to DOM events. This is an empty
114      * function by default.
115      */
116     bindEvents: function () { },
117 
118     /**
119      * @method
120      * @desc Do all expensive DOM operations that could break
121      * listeners (hiding etc.). This is an empty
122      * function by default.
123      *
124      * @param el Receives the el of this View as
125      * a parameter (needed by mixins).
126      */
127     redraw: function (el) { },
128 
129     /**
130      * @method
131      * @desc Set styles, calculates values, etc. This is called
132      * each time the View is rendered, so don't do anything expensive
133      * here. This is an empty function by default.
134      */
135     refresh: function () { },
136 
137     /**
138      * @method
139      * @desc Remove from DOM with the prospect of reattaching.
140      * Default behavior calls $(this.el).detach();
141      */
142     hide: function () {
143         $(this.el).detach();
144     },
145 
146     /**
147      * @method
148      * @desc Destroy the View beyond restoring. $By default,
149      * $(this.el).remove(); is called.
150      */
151     dispose: function () {
152         $(this.el).remove();
153     }
154 
155 });
