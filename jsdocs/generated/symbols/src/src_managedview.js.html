1 /**
  2  * @augments Backbone.View
  3  * @class A more sophisticated Backbone.View that adds
  4  * support for component lifecycles, as often needed
  5  * in RIA applications.<br /><br />
  6  *
  7  * Ribs.js uses the model ribsUIModels to store
  8  * an arbitrary number of models that the View renders.
  9  * The default ones are:<br /><br />
 10  *
 11  * data:        The default Backbone property model.<br />
 12  * dataUI:      A model added to the data model with
 13  *              Ribs.augmentModelWithUIProperties.<br />
 14  * internal:    UI model that only this View uses internally.
 15  *
 16  * @param options Options for this View.
 17 **/
 18 Ribs.ManagedView = Backbone.View.extend(
 19 /** @lends Ribs.ManagedView */
 20 {
 21 
 22     /**
 23      * @field
 24      * @desc When calling render(), if this flag is set,
 25      * a redraw will occur. When redraw is finished, this
 26      * flag is again reset to false.
 27      */
 28     invalidated: true,
 29 
 30     /**
 31      * @field
 32      * @desc Is set when the component has finished initializing
 33      * and is ready to be rendered.
 34      */
 35     initialized: false,
 36 
 37     /**
 38      * @method
 39      * @desc Override to Backbone.View.initialize; Is called on
 40      * creation, using the "new" operator.
 41      */
 42     initialize: function () {
 43         _.bindAll(this, "customInitialize", "bindToModel", "render", "unbindEvents", "redraw", "refresh", "bindEvents", "hide", "dispose");
 44         this.ribsUIModels = new Backbone.Model({ internal: new Backbone.Model() });
 45         this.ribsUIModels.set(this.options.ribsUIModels);
 46         Backbone.View.prototype.initialize.apply(this, arguments);
 47         if (this.model) {
 48             this.bindToModel(this.model);
 49         }
 50         this.customInitialize();
 51         this.initialized = true;
 52         this.render();
 53     },
 54 
 55     /**
 56      * @method
 57      * @desc Is called after the "real" initialize is completed,
 58      * apart from setting initialized flag and rendering. Without
 59      * the initialized flag the component will not render.<br /><br />
 60      *
 61      * This is a clean place for generic initialization code.
 62      * A call to this method is not expected of an overriding method.
 63      */
 64     customInitialize: function () { },
 65 
 66     /**
 67      * @method
 68      * @desc Backbone convention for setting/changing the model
 69      * for a View. Here is it is defined explicitly since the View
 70      * life cycle requires it. A possible old model is unbound of
 71      * its change handler, and the new one is in turn bound.<br /><br />
 72      *
 73      * If the new model does not yet augmented with UI attributes,
 74      * Ribs.augmentModelWithUIAttributes does this.<br /><br />
 75      *
 76      * The default models "data" and "dataUI" are set to ribsUIModels.
 77      * 
 78      * @param model The Backbone.Model we want this View to represent.
 79      */
 80     bindToModel: function (model) {
 81         if (this.model && this.model.ribsUI) {
 82             this.model.ribsUI.unbind("change", this.render);
 83         }
 84         this.model = model;
 85         if (this.model) {
 86             Ribs.augmentModelWithUIAttributes(this.model);
 87             this.model.ribsUI.bind("change", this.render);
 88         }
 89         this.ribsUIModels.set({
 90             data: model,
 91             dataUI: model.ribsUI
 92         });
 93         this.invalidated = true;
 94     },
 95 
 96     /**
 97      * @method
 98      * @desc Override to Backbone.View.render.<br /><br />
 99      *
100      * Exits if initialized flag is not set.<br /><br />
101      *
102      * Maintains the View life cycle of rendering. First
103      * unbinds all events, then redraws if invalidated flag
104      * is set, and after that calls refresh and binds all events.
105      */
106     render: function () {
107         if (!this.initialized) { return; }
108         this.unbindEvents();
109         if (this.invalidated) {
110             this.redraw(this.el);
111             this.invalidated = false;
112         }
113         this.refresh();
114         this.bindEvents();
115     },
116 
117     /**
118      * @method
119      * @desc Clear all listeners to DOM events. The default
120      * implementation only calls $(this.el).unbind().
121      */
122     unbindEvents: function () {
123         $(this.el).unbind();
124     },
125 
126     /**
127      * @method
128      * @desc Set all listeners to DOM events. This is an empty
129      * function by default.
130      */
131     bindEvents: function () { },
132 
133     /**
134      * @method
135      * @desc Do all expensive DOM operations that could break
136      * listeners (hiding etc.). This is an empty
137      * function by default.
138      */
139     redraw: function (el) { },
140 
141     /**
142      * @method
143      * @desc Set styles, calculates values, etc. This is called
144      * each time the View is rendered, so don't do anything expensive
145      * here. This is an empty function by default.
146      */
147     refresh: function () { },
148 
149     /**
150      * @method
151      * @desc Remove from DOM with the prospect of reattaching.
152      * Default behavior calls $(this.el).detach();
153      */
154     hide: function () {
155         $(this.el).detach();
156     },
157 
158     /**
159      * @method
160      * @desc Destroy the View beyond restoring. $By default,
161      * $(this.el).remove(); is called.
162      */
163     dispose: function () {
164         $(this.el).remove();
165     }
166 });
