1 /**
  2  * @method
  3  * @desc Takes an input field as its element (does not create one,
  4  * though, use templates) and utilizes it to edit the defined
  5  * model attribute. Works for both original data models and the
  6  * UI models attached to them. See Ribs.mixins.mixinComposite for
  7  * documentation of the generic options for mixins.
  8  *
  9  * @param classOptions
 10  * @param classOptions.elementSelector The default elementSelector is set to
 11  * look for elements with the same name as the given attributeName.
 12  * @param classOptions.readFunction If set, the attribute value read
 13  * from the model field is processed through this function.
 14  * @param classOptions.writeFunction If set, the value read from the
 15  * input field in DOM is processed through this function before
 16  * inserting into the corresponding model.
 17  */
 18 Ribs.mixins.textValueEdit = function (classOptions) {
 19     classOptions = classOptions || {};
 20     var TextValueEditInst = function () {
 21             return _.extend({
 22                 elementSelector: classOptions.attributeName && '[name|="' + classOptions.attributeName + '"]',
 23                 readFunction: null,
 24                 writeFunction: null,
 25 
 26                 modelChanging: function () {
 27                     this.uiModel.unbind("commitEdit", this.commit);
 28                     this.uiModel.unbind("cancelEdit", this.redraw);
 29                 },
 30                 modelChanged: function () {
 31                     this.uiModel.bind("commitEdit", this.commit);
 32                     this.uiModel.bind("cancelEdit", this.redraw);
 33                 },
 34                 redraw: function () {
 35                     var value = this.myValue;
 36                     if (this.readFunction) {
 37                         value = this.readFunction(value);
 38                     }
 39                     this.el.val(value);
 40                 },
 41 
 42                 commit: function () {
 43                     if (this.el) {
 44                         var value = this.el.val(), values = {};
 45                         if (this.writeFunction) {
 46                             value = this.writeFunction(value, this.myValue);
 47                         }
 48                         if (this.attributeName) {
 49                             values[this.attributeName] = value;
 50                             this.dataModel.set(values);
 51                         } else if (this.uiAttributeName) {
 52                             values[this.uiAttributeName] = value;
 53                             this.uiModel.set(values);
 54                         }
 55                     }
 56                 }
 57             }, classOptions || {});
 58         };
 59 
 60     return TextValueEditInst;
 61 };
