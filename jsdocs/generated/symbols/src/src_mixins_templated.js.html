1 /**
  2  * @method
  3  * @desc Uses a given template to replace the contents
  4  * of its element. Gives the template the data and UI models
  5  * as json, the latter overriding the former.
  6  *
  7  * @param classOptions
  8  * @param classOptions.templateSelector Selector for the template.
  9  * Plain string that jQuery can handle.
 10  * @param classOptions.templateFunction Function that takes json as
 11  * its parameter and returns HTML. Overrides templateSelector if both
 12  * are given.
 13  * @param classOptions.className Templates do not affect the root element
 14  * of a Backbone.View, and the same applies to Ribs.js mixins. If you wish
 15  * to have a certain class for the root DOM element, set this property to
 16  * that class and it will be passed to the element.
 17  */
 18 Ribs.mixins.templated = function (classOptions) {
 19     classOptions = classOptions || {};
 20     var defaultTemplateFunction = classOptions.templateSelector && _.template($(classOptions.templateSelector).html()),
 21         TemplatedInst = function () {
 22             return _.extend({
 23                 templateSelector: null,
 24                 templateFunction: defaultTemplateFunction,
 25                 className: null,
 26 
 27                 redraw: function () {
 28                     var modelJSON = this.dataModel ? this.dataModel.toJSON() : {},
 29                         uiModelJSON = this.uiModel.toJSON(),
 30                         json = _.extend(modelJSON, uiModelJSON);
 31 
 32                     json.t = function (name) {
 33                         return this.hasOwnProperty(name) ? this[name] : "";
 34                     };
 35                     this.el.html(this.templateFunction(json));
 36                     if (this.className) {
 37                         this.el.toggleClass(this.className, true);
 38                     }
 39                 }
 40             }, classOptions);
 41         };
 42 
 43     return TemplatedInst;
 44 };
