1 /**
  2  * Similar to Ribs.mixins.TextValueEdit, but uses a select
  3  * element to offer fixed choices in a drop down. Make sure
  4  * the elementSelector of this mixins points to a select.
  5  *
  6  * @param classOptions
  7  * @param classOptions.elementSelector The default elementSelector is set to
  8  * look for elements with the same name as the given attributeName.
  9  * @param classOptions.selectOptions Options that are populated to the
 10  * select element. Use an array of { value: <value>, text: <text> } objects.
 11  */
 12 Ribs.mixins.selectEdit = function (classOptions) {
 13     classOptions = classOptions || {};
 14     var SelectEditInst = function () {
 15             return _.extend({
 16                 elementSelector: classOptions.attributeName && '[name|="' + classOptions.attributeName + '"]',
 17                 selectOptions: [],
 18 
 19                 modelChanging: function () {
 20                     this.uiModel.unbind("commitEdit", this.commit);
 21                     this.uiModel.unbind("cancelEdit", this.redraw);
 22                 },
 23                 modelChanged: function () {
 24                     this.uiModel.bind("commitEdit", this.commit);
 25                     this.uiModel.bind("cancelEdit", this.redraw);
 26                 },
 27                 redraw: function () {
 28                     if (this.el.is("select")) {
 29                         this.selectEl = this.el;
 30                     } else {
 31                         if (this.selectEl) { this.selectEl.remove(); }
 32                         this.selectEl = $("<select></select>");
 33                         this.el.append(this.selectEl);
 34                     }
 35                     if (this.dataModel) {
 36                         var val = this.dataModel.get(this.attributeName);
 37                         _.each(this.selectOptions, _.bind(function (option) {
 38                             var optionEl = $('<option></option>');
 39                             optionEl
 40                                     .attr("value", option.value)
 41                                     .text(option.text);
 42                             if (option.value === val) {
 43                                 optionEl.attr("selected", "selected");
 44                             }
 45                             this.selectEl.append(optionEl);
 46                         }, this));
 47                     }
 48                 },
 49 
 50                 commit: function () {
 51                     if (this.selectEl) {
 52                         var value = this.selectEl.val(), values = {};
 53                         values[this.attributeName] = value;
 54                         this.dataModel.set(values);
 55                     }
 56                 }
 57             }, classOptions || {});
 58         };
 59 
 60     return SelectEditInst;
 61 };
