  5 Ribs.support.mixins.methodInherit = {
  6     inheritingMethods: [],
  7     inheritingProperties: ["inheritingMethods"],
  8 
  9     mixinInitialize: function () {
 10         this.callAllMixins("mixinInitialize", arguments);
 11         this.inheritMethods();
 12     },
 13     callAllMixins: function (methodName, originalArguments) {
 14         _.each(this.children, function (mixin) {
 15             if (typeof(mixin[methodName]) === "function") {
 16                 mixin[methodName].apply(mixin, originalArguments);
 17             }
 18         });
 19     },
 20     inheritMethods: function () {
 21         var that = this;
 22         this._inheritedMethods = this._inheritedMethods || [];
 23 
 24         _.each(this.inheritingMethods, function (methodName) {
 25             if (_.indexOf(that._inheritedMethods, methodName) === -1) {
 26                 var oldMethod = that[methodName];
 27                 that[methodName] = function () {
 28                     if (typeof(oldMethod) === "function") {
 29                         oldMethod.apply(this, arguments);
 30                     }
 31                     this.callAllMixins(methodName, arguments);
 32                 };
 33                 that._inheritedMethods.push(methodName);
 34             }
 35         });
 36     }
 37 };
