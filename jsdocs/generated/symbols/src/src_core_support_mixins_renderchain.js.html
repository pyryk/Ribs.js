1 /**
  2  * @class Creates a more sophisticated component life cycle
  3  * inside the render function.
  4  * @requires Ribs.support.mixins.methodInherit
  5 **/
  6 Ribs.support.mixins.renderChain = {
  7     /**
  8      * @field
  9      * @desc When calling render(), if this flag is set,
 10      * a redraw will occur. When redraw is finished, this
 11      * flag is again reset to false.
 12      * @default true
 13      */
 14     invalidated: true,
 15 
 16     /**
 17      * @field
 18      * @desc This flag should be set to true externally
 19      * when the component has finished initializing.
 20      * @default false
 21      */
 22     initialized: false,
 23 
 24     /**
 25      * @field
 26      * @default ["unbindEvents", "bindEvents", "redraw", "refresh", "hide", "dispose"]
 27      */
 28     inheritingMethods: ["unbindEvents", "bindEvents", "redraw", "refresh", "hide", "dispose"],
 29 
 30     /**
 31      * @method
 32      * @desc Override to Backbone.View.render.<br /><br />
 33      *
 34      * Exits if initialized flag is not set.<br /><br />
 35      *
 36      * Maintains the View life cycle of rendering. If invalidated
 37      * flag is set, unbinds all DOM events, calls redraw, and then
 38      * binds the events. After this comes always refresh,
 39      * regardless of the invalidated flag.
 40      */
 41     render: function () {
 42         if (!this.initialized) { return; }
 43         if (this.invalidated) {
 44             this.unbindEvents();
 45             this.redraw();
 46             this.bindEvents();
 47             this.invalidated = false;
 48         }
 49         this.refresh();
 50     },
 51 
 52     /**
 53      * @method
 54      * @desc Clear all listeners to DOM events. This is an empty
 55      * function by default.
 56      */
 57     unbindEvents: function () { },
 58 
 59     /**
 60      * @method
 61      * @desc Set all listeners to DOM events. This is an empty
 62      * function by default.
 63      */
 64     bindEvents: function () { },
 65 
 66     /**
 67      * @method
 68      * @desc Do all expensive DOM operations that could break
 69      * listeners (hiding etc.). This is an empty
 70      * function by default.
 71      */
 72     redraw: function () { },
 73 
 74     /**
 75      * @method
 76      * @desc Set styles, calculates values, etc. This is called
 77      * each time the View is rendered, so don't do anything expensive
 78      * here. This is an empty function by default.
 79      */
 80     refresh: function () { },
 81 
 82     /**
 83      * @method
 84      * @desc Remove from DOM with the prospect of reattaching.
 85      */
 86     hide: function () { },
 87 
 88     /**
 89      * @method
 90      * @desc Destroy the View beyond restoring.
 91      */
 92     dispose: function () { }
 93 };
