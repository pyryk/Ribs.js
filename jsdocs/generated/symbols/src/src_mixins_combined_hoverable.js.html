1 /**
  2  * @method
  3  * @desc Binds to mouseenter and mouseleave to know when
  4  * the user is hovering over the element of this mixin.
  5  * Toggles the "hovering" property of the UI model and
  6  * assigns class "hovering" to the element, if the user
  7  * is hovering.
  8  *
  9  * @param classOptions
 10  * @param classOptions.elementSelector This is the only
 11  * property that has any effect.
 12  */
 13 Ribs.mixins.hoverable = function (classOptions) {
 14     var HoverableInst = Ribs.mixins.mixinComposite(_.extend({
 15         mixinClasses: [
 16             Ribs.mixins.toggleAttribute({
 17                 onEvent: "mouseenter",
 18                 offEvent: "mouseleave",
 19                 uiAttributeName: "hovering"
 20             }),
 21             Ribs.mixins.toggleableClass({
 22                 className: "hovering",
 23                 uiAttributeName: "hovering"
 24             })
 25         ]
 26     }, classOptions || {}));
 27 
 28     return HoverableInst;
 29 };
