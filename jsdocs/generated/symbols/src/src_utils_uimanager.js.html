1 /**
  2  * @method
  3  * @desc Creates a manager to which Backbone Collections
  4  * can be added. The manager will listen to ribsUI:change:selected
  5  * and ribsUI:change:hovering events, and make sure only one
  6  * item is selected/hovering at a time.<br &><br />
  7  *
  8  * The selected/hovering item can be accessed through the
  9  * viewModel that this manager internally uses to keep track of
 10  * what is selected/hovering.
 11  *
 12  * @param key Identifier for this UI manager. The created
 13  * UI manager can be accessed through Ribs.uiManager.<key>.
 14  * @param myOptions Do not use for now.
 15  */
 16 Ribs.createUIManager = function (key, myOptions) {
 17     myOptions = myOptions || {};
 18 
 19     Ribs.uiManagers = Ribs.uiManagers || {};
 20 
 21     Ribs.uiManagers[key] = (function () {
 22         var allowMultiselect = myOptions.allowMultiselect,
 23             viewModel = new Backbone.Model({ nowHovering: null, nowSelected: null }),
 24             hoveringChanged = function (event) {
 25                 var item = event[0];
 26                 if (item === viewModel.get("nowHovering") && !item.get("hovering")) {
 27                     viewModel.set({ nowHovering: null });
 28                 } else if (item !== viewModel.get("nowHovering") && item.get("hovering")) {
 29                     var lastHovering = viewModel.get("nowHovering");
 30                     viewModel.set({ nowHovering: item });
 31                     if (lastHovering) {
 32                         lastHovering.set({ hovering: false });
 33                     }
 34                 }
 35             },
 36             selectedChanged = function (event) {
 37                 var item = event[0];
 38                 if (item === viewModel.get("nowSelected") && !item.get("selected")) {
 39                     viewModel.set({ nowSelected: null });
 40                 } else if (item !== viewModel.get("nowSelected") && item.get("selected")) {
 41                     var lastSelected = viewModel.get("nowSelected");
 42                     viewModel.set({ nowSelected: item });
 43                     if (!allowMultiselect && lastSelected) {
 44                          lastSelected.set({ selected: false });
 45                     }
 46                 }
 47             },
 48             unregister = function (model) {
 49                 if (model) {
 50                     model.unbind("ribsUI:change:hovering", hoveringChanged);
 51                     model.unbind("ribsUI:change:selected", selectedChanged);
 52                 }
 53             },
 54             register = function (model) {
 55                 if (model) {
 56                     unregister(model);
 57                     model.bind("ribsUI:change:hovering", hoveringChanged);
 58                     model.bind("ribsUI:change:selected", selectedChanged);
 59                 }
 60             };
 61 
 62         return {
 63             /**
 64              * @method
 65              * @desc Registers the given Backbone Model/Collection to be
 66              * managed by this manager.
 67              *
 68              * @param The Model or Collection wanted to be managed.
 69              */
 70             register: register,
 71 
 72             /**
 73              * @method
 74              * @desc Stops managing the given Model/Collection.
 75              *
 76              * @param The Model or Collection wanted to be unregistered.
 77              */
 78             unregister: unregister,
 79 
 80             /**
 81              * @method
 82              * @desc Returns the viewModel that contains the state of
 83              * this manager.
 84              */
 85             getViewModel: function () { return viewModel; }
 86         };
 87     }());
 88 };
