 17     mixinInitialize: function () {

 19         this.children = [];

 21     },

 23         var that = this,
 24             mixin = _.extend({}, mixinDef);
 25         _.each(this.inheritingProperties, function (value) {
 26             mixin[value] = that[value];
 27         });
 28         _.bind(function () { _.bindAll(this); }, mixin)();
 29         this.children.push(mixin);
 30     },

 33             if (this.childrenTypes[i].elementSelector === elementSelector) {
 34                 return this.childrenTypes[i];
 35             }
 36         }
 37         return null;
 38     }
