1 /**
  2  * @method
  3  * @desc Observes attributes of the data and UI models and
  4  * sets the invalidated flag of the parent View if they change.
  5  *
  6  * @param classOptions
  7  * @param classOptions.excludedAttributes Array of attribute names
  8  * which should not be observed in the original data model (even if
  9  * defined in includedAttributes).
 10  * @param classOptions.includedAttributes Array of attribute names
 11  * which should be observed in the original data model.
 12  * @param classOptions.excludedRibsUIAttributes Same as excludedAttributes,
 13  * except for observing changes in the UI model that is attached
 14  * to the original data model.
 15  * @param classOptions.includedRibsUIAttributes Same as excludedAttributes,
 16  * except for observing changes in the UI model that is attached
 17  * to the original data model.
 18  */
 19 Ribs.mixins.invalidateOnChange = function (classOptions) {
 20     var InvalidateOnChangeInst = function (parent) {
 21             return _.extend({
 22                 excludedAttributes: null,
 23                 includedAttributes: null,
 24                 excludedRibsUIAttributes: null,
 25                 includedRibsUIAttributes: null,
 26 
 27                 modelChanging: function () {
 28                     if (this.dataModel) {
 29                         this.dataModel.unbind("change", this.change);
 30                     }
 31                     if (this.uiModel.safeUnbind) {
 32                         this.uiModel.safeUnbind("change", this.ribsUIChange);
 33                     }
 34                 },
 35                 modelChanged: function () {
 36                     if (this.dataModel) {
 37                         this.dataModel.bind("change", this.change);
 38                     }
 39                     this.uiModel.bind("change", this.ribsUIChange);
 40                 },
 41                 change: function (ev) {
 42                     _.each(ev.changedAttributes(), this.checkAttribute);
 43                 },
 44                 checkAttribute: function (value, attrName) {
 45                     var excluded = this.excludedAttributes && _.indexOf(this.excludedAttributes, attrName) !== -1,
 46                         included = this.includedAttributes && _.indexOf(this.includedAttributes, attrName) !== -1;
 47                     if (!excluded && included && !parent.invalidated) {
 48                         parent.invalidated = true;
 49                         _.defer(parent.render);
 50                     }
 51                 },
 52                 ribsUIChange: function (ev) {
 53                     _.each(ev.changedAttributes(), this.checkUIAttribute);
 54                 },
 55                 checkUIAttribute: function (value, attrName) {
 56                     var excluded = this.excludedRibsUIAttributes && _.indexOf(this.excludedRibsUIAttributes, attrName) !== -1,
 57                         included = this.includedRibsUIAttributes && _.indexOf(this.includedRibsUIAttributes, attrName) !== -1;
 58                     if (!excluded && included && !parent.invalidated) {
 59                         parent.invalidated = true;
 60                         _.defer(parent.render);
 61                     }
 62                 }
 63             }, classOptions || {});
 64         };
 65 
 66     return InvalidateOnChangeInst;
 67 };
