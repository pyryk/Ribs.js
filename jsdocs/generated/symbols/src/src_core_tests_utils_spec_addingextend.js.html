1 describe("Ribs.utils.compose", function () {
  2     var obj;
  3 
  4     beforeEach(function () {
  5         obj = { a: "1", b: "2", c: "3", e: ["firstItem", "secondItem"] };
  6     });
  7 
  8     it("Should replace properties from additional sources", function () {
  9         Ribs.utils.compose(obj, { a: "5", d: "2" });
 10         expect(obj.a).toEqual("5");
 11         expect(obj.b).toEqual("2");
 12         expect(obj.c).toEqual("3");
 13         expect(obj.d).toEqual("2");
 14     });
 15 
 16     it("Should throw an error when trying to replace an array with something else", function () {
 17         var errorThrown = false;
 18         try {
 19             Ribs.utils.compose(obj, { e: "Hello world" });
 20         } catch(e) {
 21             errorThrown = e === "addingExtendArrayWithNonArray";
 22         }
 23         expect(errorThrown).toBeTruthy();
 24     });
 25 
 26     it("Should concatenate arrays", function () {
 27         Ribs.utils.compose(obj, { e: ["thirdItem"] });
 28         expect(obj.e.length).toEqual(3);
 29         expect(obj.e[2]).toEqual("thirdItem");
 30     });
 31 
 32     it("Should throw an error when trying to replace a function with something else", function () {
 33         var errorThrown = false;
 34         obj.f = function () { };
 35         try {
 36             Ribs.utils.compose(obj, { f: null });
 37         } catch(e) {
 38             errorThrown = e === "addingExtendFunctionWithNonFunction";
 39         }
 40         expect(errorThrown).toBeTruthy();
 41     });
 42 
 43     it("Should call all replaced functions", function () {
 44         var f1, f1Called, f2, f2Called;
 45 
 46         f1Called = false;
 47         f2Called = false;
 48 
 49         f1 = function () {
 50             if (f2Called) {
 51                 throw "Wrong order in calling extended function (old one should be first).";
 52             }
 53             f1Called = true;
 54         };
 55 
 56         f2 = function () {
 57             f2Called = true;
 58         };
 59         
 60         obj.f = f1;
 61         Ribs.utils.compose(obj, { f: f2 });
 62         obj.f();
 63         expect(f1Called).toBeTruthy();
 64         expect(f2Called).toBeTruthy();
 65     });
 66 });
