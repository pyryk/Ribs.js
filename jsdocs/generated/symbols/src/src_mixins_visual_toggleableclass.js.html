1 /**
  2  * @method
  3  * @desc Toggle a class to the element depending on the boolean
  4  * value of the associated attribute.<br &><br />
  5  *
  6  * Does not observe the changes in the model, so you may have to
  7  * use invalidateonchange mixin.
  8  *
  9  * @param classOptions
 10  * @param classOptions.className Class to give to the element if
 11  * the attribute is true.
 12  * @param classOptions.inverse Give element the class when the
 13  * attribute is false, and remove it when it is true.
 14  */
 15 Ribs.mixins.toggleableClass = function (classOptions) {
 16     classOptions = classOptions || {};
 17     var ToggleableClassInst = function () {
 18             return _.extend({
 19                 className: classOptions.attributeName || classOptions.uiAttributeName,
 20                 inverse: false,
 21                 refresh: function () {
 22                     var value = this.myValue;
 23                     if (this.inverse) {
 24                         value = !value;
 25                     }
 26                     if (this.el.hasClass(this.className) !== value) {
 27                         this.el.toggleClass(this.className, value);
 28                     }
 29                 }
 30             }, classOptions || {});
 31         };
 32 
 33     return ToggleableClassInst;
 34 };
