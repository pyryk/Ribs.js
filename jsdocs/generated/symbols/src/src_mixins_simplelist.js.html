1 /**
  2  * @method
  3  * @desc Most complicated of the standard mixins, but this could
  4  * not be avoided. Creates a list of a defined Backbone.Collection,
  5  * binds to changes in it, and updates itself accordingly.<br /><br />
  6  *
  7  * If item are added dynamically one-by-one, they are inserted at the
  8  * bottom of the list. If the order is crucial while adding new ones,
  9  * invalidate the parent View to call redraw when rendering.
 10  *
 11  * @param classOptions
 12  * @param classOptions.ItemRenderer Backbone.View class to instantiate
 13  * for every item on the list.
 14  * @param classOptions.itemTagName Name of the tag of the new list items.
 15  * This has to be defined in the list itself because the list items cannot
 16  * affect the name of their own element tag after creation. This is
 17  * because of the design of Backbone.View.
 18  * @param classOptions.itemClassName Same as itemTagName, except that
 19  * this one is set as the class of each new list item element.
 20  */
 21 Ribs.mixins.simpleList = function (classOptions) {
 22     classOptions = classOptions || {};
 23     var ItemRenderer = classOptions.ItemRenderer,
 24         SimpleListInst = function (parent) {
 25             var listModel, listViews, refreshingList;
 26             return _.extend({
 27                 itemTagName: null,
 28                 itemClassName: null,
 29 
 30                 modelChanging: function () {
 31                     _.each(listViews, function (view) {
 32                         view.dispose();
 33                     });
 34                     listViews = {};
 35                     if (listModel) {
 36                         listModel.unbind("add", this.addOne);
 37                         listModel.unbind("remove", this.removeOne);
 38                         listModel.unbind("refresh", this.addAll);
 39                     }
 40                 },
 41                 modelChanged: function () {
 42                     listModel = this.myValue ? this.myValue : this.dataModel;
 43                     if (listModel) {
 44                         listModel.bind("add", this.addOne);
 45                         listModel.bind("remove", this.removeOne);
 46                         listModel.bind("refresh", this.addAll);
 47                         this.addAll();
 48                     }
 49                 },
 50                 redraw: function () {
 51                     this.el.children().detach();
 52                     _.each(listViews, _.bind(function (view) {
 53                         this.el.append(view.el);
 54                     }, this));
 55                 },
 56                 render: function () {
 57                     _.each(listViews, function (view) {
 58                         view.render();
 59                     });
 60                 },
 61 
 62                 addOne: function (item) {
 63                     if (!listViews.hasOwnProperty(item.cid)) {
 64                         var listView = new ItemRenderer({
 65                                 model: item,
 66                                 tagName: this.itemTagName,
 67                                 className: this.itemClassName
 68                             });
 69                         listViews[item.cid] = listView;
 70                         if (!refreshingList) {
 71                             parent.invalidated = true;
 72                             parent.render();
 73                         }
 74                     }
 75                 },
 76                 addAll: function () {
 77                     refreshingList = true;
 78                     listViews = {};
 79                     if (listModel.each) {
 80                         listModel.each(this.addOne);
 81                     }
 82                     parent.invalidated = true;
 83                     parent.render();
 84                     refreshingList = false;
 85                 },
 86                 removeOne: function (item) {
 87                     if (listViews[item.cid]) {
 88                         $(listViews[item.cid].el).remove();
 89                         delete listViews[item.cid];
 90                     }
 91                 }
 92             }, classOptions || {});
 93         };
 94 
 95     return SimpleListInst;
 96 };
