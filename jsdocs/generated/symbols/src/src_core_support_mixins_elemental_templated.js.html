1 /**
  2  * @class
  3  * @requires Ribs.mixins.support.dom.element
  4  */
  5 Ribs.mixins.support.dom.templated = {
  6     /**
  7      * @field
  8      * @desc jQuery selector for the template element.
  9      * @default null
 10      */
 11     templateSelector: null,
 12 
 13     /**
 14      * @field
 15      * @desc Plain string to use as the template.<br /><br />
 16      *
 17      * Overrides templateSelector.
 18      * @default null
 19      */
 20     templateString: null,
 21 
 22     /**
 23      * @field
 24      * @desc Template function to call when using the template.
 25      * Gets the model json as its first parameter.<br /><br />
 26      *
 27      * Overrides templateSelector and templateString
 28      * @default null
 29      */
 30     templateFunction: null,
 31 
 32     /**
 33      * @field
 34      * @desc If set to true, will replace the el property with the
 35      * template element. Otherwise replaces the contents of the el
 36      * with the templated element, leaving the root tag.<br /><br />
 37      *
 38      * Notice that you will have to re-append this el to DOM manually
 39      * in case it is overwritten.<br /><br />
 40      * @default false
 41      */
 42     overwriteEl: false,
 43 
 44     /**
 45      * @field
 46      * @desc "Data" to give to the template function. Should be
 47      * calculated right before rendering the template in redraw.
 48      */
 49     json: null,
 50 
 51     /**
 52      * @method
 53      * @desc Initializes the template based on the mentioned fields.
 54      */
 55     mixinInitialize: function () {
 56         if (!this.templateFunction) {
 57             if (this.templateSelector && !this.templateString) {
 58                 this.templateString = $(this.templateSelector).html();
 59             }
 60             if (this.templateString) {
 61                 this.templateFunction = _.template(this.templateString);
 62             }
 63         }
 64     },
 65 
 66     /**
 67      * @method
 68      * @desc This is where the template is applied.
 69      */
 70     redraw: function () {
 71         if (this.templateFunction) {
 72             if (this.overwriteEl) {
 73                 this.el = $(this.templateFunction(this.json || {}));
 74             } else {
 75                 $(this.el).html(this.templateFunction(this.json || {}));
 76             }
 77         }
 78     }
 79 };
