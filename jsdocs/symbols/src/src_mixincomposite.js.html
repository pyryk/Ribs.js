 34                     }

 88                                 mixin.el.bind(eventName, method);

103                         }
104                     }
105                     _.each(this.mixins, _.bind(function (mixin) {
106                         updateMixinEl(mixin, this.el);
107                         if (mixin.redraw) {

109                         }
110                     }, this));
