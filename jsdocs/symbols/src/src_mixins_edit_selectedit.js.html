1 Ribs.mixins.selectEdit = function (classOptions) {
  2     classOptions = classOptions || {};
  3     var SelectEditInst = function () {
  4             var model;
  5             return _.extend({
  6                 modelName: "data",
  7                 attributeName: null,
  8                 elementSelector: classOptions.attributeName && '[name|="' + classOptions.attributeName + '"]',
  9                 selectOptions: [],
 10 
 11                 modelChanging: function () {
 12                     if (model) {
 13                         model.unbind("commitEdit", this.commit);
 14                         model.unbind("cancelEdit", this.redraw);
 15                     }
 16                 },
 17                 modelChanged: function () {
 18                     model = this.getMyModel();
 19                     if (model) {
 20                         model.bind("commitEdit", this.commit);
 21                         model.bind("cancelEdit", this.redraw);                        
 22                     }
 23                 },
 24                 redraw: function () {
 25                     if (this.el.is("select")) {
 26                         this.selectEl = this.el;
 27                     } else {
 28                         if (this.selectEl) { this.selectEl.remove(); }
 29                         this.selectEl = $("<select></select>");
 30                         this.el.append(this.selectEl);
 31                     }
 32 
 33                     var val = this.getMyValue();
 34                     _.each(this.selectOptions, _.bind(function (option) {
 35                         var optionEl = $('<option></option>');
 36                         optionEl
 37                                 .attr("value", option.value)
 38                                 .text(option.text);
 39                         if (option.value === val) {
 40                             optionEl.attr("selected", "selected");
 41                         }
 42                         this.selectEl.append(optionEl);
 43                     }, this));
 44                 },
 45 
 46                 commit: function () {
 47                     if (this.selectEl) {
 48                         this.setMyValue(this.selectEl.val());
 49                     }
 50                 }
 51             }, Ribs.mixinHelpers, classOptions || {});
 52         };
 53 
 54     return SelectEditInst;
 55 };
