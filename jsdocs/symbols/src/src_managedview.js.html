1 // A more sophisticated Backbone.View that adds
  2 // support for component lifecycles, as often needed
  3 // in RIA applications.
  4 //
  5 // Ribs.js uses the model ribsUIModels to store
  6 // an arbitrary number of models that the View renders.
  7 // The default ones are:
  8 //
  9 // data:        The default Backbone property model.
 10 // dataUI:      A model added to the data model with
 11 //              Ribs.augmentModelWithUIProperties.
 12 // internal:    UI model that only this View uses internally.
 13 //
 14 
 15 Ribs.ManagedView = Backbone.View.extend({
 16     invalidated: true,
 17 
 18     initialize: function () {
 19         _.bindAll(this, "customInitialize", "bindToModel", "render", "unbindEvents", "redraw", "refresh", "bindEvents", "hide", "dispose");
 20         this.ribsUIModels = new Backbone.Model({ internal: new Backbone.Model() });
 21         this.ribsUIModels.set(this.options.ribsUIModels);
 22         Backbone.View.prototype.initialize.apply(this, arguments);
 23         if (this.model) {
 24             this.bindToModel(this.model);
 25         }
 26         this.customInitialize();
 27         this.initialized = true;
 28         this.render();
 29     },
 30     customInitialize: function () { },
 31     bindToModel: function (model) {
 32         if (this.model && this.model.ribsUI) {
 33             this.model.ribsUI.unbind("change", this.render);
 34         }
 35         this.model = model;
 36         if (this.model) {
 37             Ribs.augmentModelWithUIAttributes(this.model);
 38             this.model.ribsUI.bind("change", this.render);
 39         }
 40         this.ribsUIModels.set({
 41             data: model,
 42             dataUI: model.ribsUI
 43         });
 44         this.invalidated = true;
 45     },
 46     render: function () {
 47         if (!this.initialized) { return; }
 48         this.unbindEvents();
 49         if (this.invalidated) {
 50             this.redraw(this.el);
 51             this.invalidated = false;
 52         }
 53         this.refresh();
 54         this.bindEvents();
 55     },
 56     unbindEvents: function () {
 57         $(this.el).unbind();
 58     },
 59     bindEvents: function () { },
 60     redraw: function (el) { },
 61     refresh: function () { },
 62     hide: function () {
 63         $(this.el).detach();
 64     },
 65     dispose: function () {
 66         $(this.el).remove();
 67     }
 68 });
