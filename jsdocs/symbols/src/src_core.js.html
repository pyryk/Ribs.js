 11 var Ribs = {};
 12 
 13 Ribs.VERSION = "0.0.84";
 14 
 15 Ribs.mixins = {};
 16 
 17 Ribs.mixinMethods = [
 18     "customInitialize",
 19     "bindToModel",
 20     "render", "redraw", "refresh",
 21     "unbindEvents", "bindEvents",
 22     "hide", "dispose"
 23 ];
 24 
 25 Ribs.inheritingMixinProperties = [
 26     "attributeName", "modelName"
 27 ];

 30 /**
 31 * Creates a Backbone.View that inherits Ribs.ManagedView
 32 * and contains the mixins defined in myOptions.
 33 **/
 34 Ribs.createMixed = function (myOptions) {
 35     myOptions = myOptions || {};
 36 
 37     var Buildee = myOptions.base ? myOptions.base.extend() : Ribs.ManagedView.extend(),
 38         NewRootMixin = Ribs.mixinParser.createMixinFromDefinitions(myOptions.mixins),
 39         OldRootMixin = myOptions.base && myOptions.base.RootMixin,
 40 
 41         delegateOneToRootMixin = function (methodName) {
 42             Buildee.prototype[methodName] = function () {
 43                 Ribs.ManagedView.prototype[methodName].apply(this, arguments);
 44                 if (this.rootMixin && this.rootMixin[methodName]) {
 45                     this.rootMixin[methodName].apply(this.rootMixin, arguments);
 46                 }
 47             };
 48         };
 49 
 50     _.each(Ribs.mixinMethods, delegateOneToRootMixin);
 51 
 52     if (OldRootMixin) {
 53         Buildee.RootMixin = Ribs.mixins.mixinComposite({
 54             mixinClasses: [OldRootMixin, NewRootMixin]
 55         });
 56     } else {
 57         Buildee.RootMixin = NewRootMixin;
 58     }
 59 
 60     Buildee.prototype.initialize = function () {
 61         if (typeof(Buildee.RootMixin) === "function") {
 62             this.rootMixin = new Buildee.RootMixin(this, this.options.model);
 63         } else {
 64             this.rootMixin = _.extend({}, Buildee.RootMixin);
 65         }
 66         Ribs.ManagedView.prototype.initialize.apply(this, arguments);
 67     };
 68 
 69     return Buildee;
 70 };
